VUE基础语法

1.环境

调用bootcdn的js，简化版，min.js,做调试/学习使用。*bootcdn：https://www.bootcdn.cn*

<!DOCTYPE html>
<html>
<head>
	<title></title>
<style>
	.bg{
		color: blue;
	}
</style>
<script src="https://cdn.bootcss.com/vue/2.6.10/vue.min.js"></script>//直接引用
</head>
<body>
<div class="bg">
	1.My first Vue!!
	{{msg}}
</div>
<script>
	new Vue({
		el:'.bg',
		data:{
			msg:'2.Hello!'
		}
	})
</script>
</body>
</html>

2.绑定

两个div绑定一个div：

（绑定其一id为app）

<!DOCTYPE html>
<html>
<head>
	<title></title>
<style>
	.bg{
		color: blue;
	}
</style>
<script src="https://cdn.bootcss.com/vue/2.6.10/vue.min.js"></script>
</head>
<body>

<div class="bg">
	1.My first Vue!!
	{{msg}}
</div>

<div class="bg" id="app">
	1.My Second Vue!!
	{{msg}}
</div>

<script>
	new Vue({
		el:'#app',
		data:{
			msg:'2.Hello!'
		}
	})
</script>
</body>
</html>

</div>

3.模板

1.变量的声明，调用，计算

<div id="app">
	<div class="bg">
		1.My first Vue!!
		{{msg}}
	</div>
	<div class="bg">
		1.My Second Vue!!
		{{msg}}
	</div>
	{{count}}//调用
    {{ ( count + 1 )*10 }}//计算
</div>

<script>
	new Vue({
		el:'#app',
		data:{
			msg:'2.Hello!',	//注意添加后，不要遗忘逗号
			count:0	//声明变量，在data里
		}
	})
</script>

输出：

2.template输出

<div id="app">
	<div class="bg">
		1.My first Vue!!
		{{msg}}
	</div>

		1.My Second Vue!!
		{{msg}}
	</div>
	
	{{count}}<br>

	{{ ( count + 1 )*10 }}<br>

	{{template}}<br>//原始调用

	</div>

3.,a标签与 div id 绑定

<div id="app">
	<div class="bg" v-bind:id='bg1'>
		1.My first Vue!!
		{{msg}}
	</div>
	<div class="bg">
		1.My Second Vue!!
		{{msg}}
	</div>

	{{count}}<br>

	{{ ( count + 1 )*10 }}<br>

	{{template}}<br>

	<div v-html='template'>

	</div>

	<a href="https://www.baidu.com">百度</a>
	<a v-bind:href="url">百度</a>

</div>

<script>
	new Vue({
		el:'#app',
		data:{
			bg1:'app-bind',
			msg:'2.Hello!',
			count:0,
			template:'<div>This is template !</div>',
			url:'https://www.baidu.com'
		}
	})
</script>

输出：

4.按钮绑定

<div id="app">
	<div class="bg" v-bind:id='bg1'>
		1.My first Vue!!
		{{msg}}
	</div>
	<div class="bg">
		1.My Second Vue!!
		{{msg}}
	</div>

	{{count}}<br>

	{{ ( count + 1 )*10 }}<br>

	{{template}}<br>

	<div v-html='template'>

	</div>

	<a href="https://www.baidu.com">百度</a>
	<a v-bind:href="url">百度</a>

	<button type="button" v-on:click="submit()">增加</button>

</div>

<script>
	new Vue({
		el:'#app',
		data:{
			bg1:'app-bind',
			msg:'2.Hello!',
			count:0,
			template:'<div>This is template !</div>',
			url:'https://www.baidu.com'
		},methods:{
			submit:function(){
				this.count++
			}
		}

	})
</script>
